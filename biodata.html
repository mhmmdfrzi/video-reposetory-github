Coba tebak!") # Memilih angka secara acak antara 1 dan 100 angka_rahasia = random.randint(1, 100) tebakan_benar = False jumlah_tebakan = 0 while not tebakan_benar: tebakan = int(input("Masukkan tebakan Anda: ")) jumlah_tebakan += 1 if tebakan < angka_rahasia: print("Tebakan Anda terlalu rendah. Coba lagi!") elif tebakan > angka_rahasia: print("Tebakan Anda terlalu tinggi. Coba lagi!") else: print(f"Selamat! Anda berhasil menebak angka {angka_rahasia} dengan benar!") print(f"Anda menebak dalam {jumlah_tebakan} kali tebakan.") tebakan_benar = True print("Terima kasih telah bermain!") if __name__ == "__main__": main()
